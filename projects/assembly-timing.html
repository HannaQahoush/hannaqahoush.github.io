Assembly Timing Game

A reaction-based timing game built entirely in 8086 Assembly. The player mentally counts a randomly shown duration, then presses spacebar — and the program rates their accuracy.

Project Overview

The game presents the player with a random target time between 0 and 30 seconds. After pressing spacebar to begin, a background timer starts running in the CPU. The player then presses spacebar a second time when they believe the target duration has elapsed. The program calculates the actual elapsed time, compares it to the target, and displays an accuracy rating.

The entire program is written in 8086 Assembly using only low-level CPU instructions — no standard libraries. Timing is handled using hardware interrupt loops, and all I/O is done via DOS interrupts.

Source Code

org 100h   

.data   
randVal db ? 
userTime db ?
str0 db "Welcome to the Timing Guessing Game press [Space] to start !$"
str1 db "The target time: $"
str2 db "The User's time: $"
str3 db "Difference between guessed time and target time: $" 
str4 db "SUPER$"
str5 db "GOOD$"
str6 db "TRY HARDER NEXT TIME$"
str7 db "Press enter to try again !$"
str8 db "Press space to start the timer !$"
startTime dw ?
endTime dw ?
diff db ? 

.code
mov ax, @data
mov ds, ax

main:  
    mov dx, offset str0      ; print welcome
    mov ah, 9
    int 21h  
    call newline
    
Start_Game:  
    mov ah, 1  
    int 21h    
    cmp al, 20h              ; compare entered value with 20h (ASCII for space)    
    JE Print_ranVal  
    JMP Start_Game  

Print_ranVal:
    call newline
    mov dx, offset str1
    mov ah, 9
    int 21h                
    mov ah, 0                ; get system time to generate random number (1-30)
    int 1Ah                  ; time saved as CX:DX = clock ticks since midnight
    mov ax, dx               ; use low bits — they change faster, more "random"
    mov dx, 0                
    mov bx, 30               
    div bx                   ; ax = quotient, dx = remainder (0-29)
    inc dx                   ; range becomes 1-30
    mov randVal, dl          
    mov al, randVal
    call print_number
    call newline
    
TimerStartWait: 
    mov dx, offset str8
    mov ah, 9
    int 21h
    mov ah, 1  
    int 21h                  
    cmp al, 20h
    JNE TimerStartWait

StartTimer:                            
    mov ah, 0
    int 1Ah
    mov startTime, dx                         
    mov ah, 1  
    int 21h                  
    cmp al, 20h
    JNE StartTimer

StopTimer:
    mov ah, 0
    int 1Ah        
    mov endTime, dx          
    mov ax, endTime
    sub ax, startTime        ; ax = tick difference                              
    mov dx, 0                
    mov cx, 18             
    div cx                   ; convert ticks to seconds (~18 ticks per second)
    
store_time:
    mov userTime, al         
    sub al, randVal
    JNS positive_diff
    neg al
    
positive_diff:
    mov diff, al
    call newline
    mov dx, offset str2
    mov ah, 9
    int 21h
    mov al, userTime
    call print_number
    call newline
    mov dx, offset str3
    mov ah, 9
    int 21h
    mov al, diff
    call print_number
    call newline
    cmp diff, 2
    JLE super
    cmp diff, 5
    JLE good
    
tryharder:
    mov dx, offset str6
    JMP print_result
    
good:
    mov dx, offset str5
    JMP print_result
    
super:
    mov dx, offset str4
    
print_result:
    mov ah, 9
    int 21h
    
EndGame:
    call newline
    mov dx, offset str7
    mov ah, 9
    int 21h
    mov ah, 1
    int 21h
    cmp al, 0Dh
    JE Print_ranVal          
    
done:
    ret

; ── Helper Procedures ──────────────────────────────

newline:
    mov dl, 0ah
    mov ah, 02h
    int 21h
    mov dl, 0dh
    mov ah, 02h
    int 21h
    ret

print_number: 
    cmp al, 10
    JL print_one_digit
    mov ah, 0
    mov bl, 10
    div bl
    add ax, 3030h
    mov dl, al
    mov dh, ah
    mov ah, 2
    int 21h
    mov dl, dh
    int 21h
    ret
    
print_one_digit:
    add al, '0'
    mov dl, al
    mov ah, 2
    int 21h
    ret
